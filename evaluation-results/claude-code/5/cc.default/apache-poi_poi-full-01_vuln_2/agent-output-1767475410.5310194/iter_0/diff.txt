+    /**
+     * Sanitizes a file name to prevent path traversal attacks.
+     * Removes any path separator characters and prevents use of ".." sequences.
+     *
+     * @param fileName the original file name from the OLE2 entry
+     * @return a sanitized file name safe for use in file operations
+     */
+    private static String sanitizeFileName(String fileName) {
+        if (fileName == null || fileName.isEmpty()) {
+            return "unnamed";
+        }
+        // Replace path separators and remove ".." to prevent path traversal
+        String sanitized = fileName.replace('/', '_')
+                                   .replace('\\', '_')
+                                   .replace("..", "__");
+        // Also trim and handle edge cases
+        sanitized = sanitized.trim();
+        if (sanitized.isEmpty() || sanitized.equals(".") || sanitized.equals("..")) {
+            return "unnamed";
+        }
+        return sanitized;
+    }

+     *
+     * @param parent the parent directory
+     * @param fileName the file name to use (will be sanitized)
+     * @return a File object guaranteed to be within the parent directory
+     * @throws IOException if the resulting path would escape the parent directory
+     */
+    private static File createSafeFile(File parent, String fileName) throws IOException {
+        String safeName = sanitizeFileName(fileName);
+        File file = new File(parent, safeName);

+        String parentCanonical = parent.getCanonicalPath();
+        String fileCanonical = file.getCanonicalPath();

+            !fileCanonical.equals(parentCanonical)) {
+            throw new IOException("Path traversal attempt detected: " + fileName);
+        }
+        return file;
+    }
