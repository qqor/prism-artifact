+    private String validateTesseractPath(String path) throws IOException {
+        if (path == null || path.isEmpty()) {
+            throw new IOException("Tesseract path cannot be null or empty");
+        }

+        // Reject paths containing shell metacharacters and command injection patterns

+    }
